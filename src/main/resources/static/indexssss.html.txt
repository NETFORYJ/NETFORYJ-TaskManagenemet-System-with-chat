<script>
    let currentUser = null;
    let jwtToken = null;
    let taskChart = null;

    function showToast(message, type) {
        const toast = document.createElement('div');
        toast.className = `toast toast-${type}`;
        toast.innerText = message;
        document.body.appendChild(toast);
        setTimeout(() => toast.remove(), 3000);
    }

    // ========== LOGIN / REGISTER ==========
    
    async function handleLogin() {
        const email = document.getElementById('loginEmail').value;
        const password = document.getElementById('loginPassword').value;
        
        showToast('Logging in...', 'info');
        
        try {
            const response = await fetch('/api/auth/login', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ email, password })
            });
            
            if (response.ok) {
                const data = await response.json();
                jwtToken = data.token;
                currentUser = data;
                localStorage.setItem('token', jwtToken);
                localStorage.setItem('user', JSON.stringify(data));
                showDashboard();
                showToast('Login successful!', 'success');
            } else {
                showToast('Login failed!', 'error');
            }
        } catch (error) {
            showToast('Error: ' + error.message, 'error');
        }
    }
    
    document.getElementById('registerForm').addEventListener('submit', async (e) => {
        e.preventDefault();
        const response = await fetch('/api/auth/register', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                fullName: document.getElementById('regFullName').value,
                email: document.getElementById('regEmail').value,
                password: document.getElementById('regPassword').value,
                tenantName: document.getElementById('regTenantName').value,
                subdomain: document.getElementById('regSubdomain').value
            })
        });
        if (response.ok) {
            showToast('Registration successful! Please login.', 'success');
            document.getElementById('registerContainer').classList.add('hidden');
            document.getElementById('loginContainer').classList.remove('hidden');
        } else {
            showToast('Registration failed!', 'error');
        }
    });
    
    document.getElementById('showRegister').onclick = () => {
        document.getElementById('loginContainer').classList.add('hidden');
        document.getElementById('registerContainer').classList.remove('hidden');
    };
    document.getElementById('showLogin').onclick = () => {
        document.getElementById('registerContainer').classList.add('hidden');
        document.getElementById('loginContainer').classList.remove('hidden');
    };
    
    function showDashboard() {
        document.getElementById('loginContainer').classList.add('hidden');
        document.getElementById('registerContainer').classList.add('hidden');
        document.getElementById('dashboardContainer').classList.remove('hidden');
        document.getElementById('userName').innerText = currentUser.fullName;
        document.getElementById('workspaceName').innerHTML = currentUser.tenantName || 'Workspace';
        loadDashboard();
        loadTasks();
        loadMyTasks();
    }
    
    function logout() {
        localStorage.removeItem('token');
        localStorage.removeItem('user');
        location.reload();
    }
    
    // Check existing session
    const savedToken = localStorage.getItem('token');
    const savedUser = localStorage.getItem('user');
    if (savedToken && savedUser) {
        jwtToken = savedToken;
        currentUser = JSON.parse(savedUser);
        showDashboard();
    }
    
    // ========== DASHBOARD ==========
    
    async function loadDashboard() {
        try {
            const response = await fetch('/api/tasks', { headers: { 'Authorization': `Bearer ${jwtToken}` } });
            const tasks = await response.json();
            const total = tasks.length;
            const available = tasks.filter(t => t.status === 'AVAILABLE').length;
            const inProgress = tasks.filter(t => t.status === 'IN_PROGRESS').length;
            const completed = tasks.filter(t => t.status === 'COMPLETED').length;
            
            document.getElementById('totalTasks').innerText = total;
            document.getElementById('availableTasks').innerText = available;
            document.getElementById('inProgressTasks').innerText = inProgress;
            document.getElementById('completedTasks').innerText = completed;
            
            if (taskChart) taskChart.destroy();
            const ctx = document.getElementById('taskChart').getContext('2d');
            taskChart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: ['Available', 'In Progress', 'Completed'],
                    datasets: [{ data: [available, inProgress, completed], backgroundColor: ['#10b981', '#f59e0b', '#3b82f6'] }]
                }
            });
        } catch (error) {
            showToast('Error loading dashboard', 'error');
        }
    }
    
    // ========== TASKS ==========
    
    async function loadTasks() {
        try {
            const response = await fetch('/api/tasks', { headers: { 'Authorization': `Bearer ${jwtToken}` } });
            const tasks = await response.json();
            const container = document.getElementById('tasksList');
            
            if (tasks.length === 0) {
                container.innerHTML = '<p class="col-span-full text-center text-gray-500">No tasks available</p>';
                return;
            }
            
            container.innerHTML = tasks.map(task => `
                <div class="task-card bg-white p-5">
                    <div class="flex justify-between items-start mb-3">
                        <h3 class="font-bold">${escapeHtml(task.title)}</h3>
                        <span class="status-badge status-${task.status}">${task.status}</span>
                    </div>
                    <p class="text-sm text-gray-600 mb-3">${escapeHtml(task.description?.substring(0, 100) || 'No description')}</p>
                    <div class="text-xs text-gray-500 mb-3">
                        <i class="fas fa-user mr-1"></i>Created by: ${escapeHtml(task.createdByName)}
                    </div>
                    ${task.status === 'AVAILABLE' ? 
                        `<button onclick="takeTask(${task.id}, this)" class="w-full bg-green-500 text-white py-2 rounded-lg text-sm hover:bg-green-600">Take Task</button>` : 
                        task.status === 'IN_PROGRESS' && task.assignedTo === currentUser.id ?
                        `<button onclick="showCompleteModal(${task.id})" class="w-full bg-blue-500 text-white py-2 rounded-lg text-sm hover:bg-blue-600">Complete Task</button>` :
                        task.status === 'COMPLETED' ?
                        `<button onclick="viewTaskDetails(${task.id})" class="w-full bg-gray-500 text-white py-2 rounded-lg text-sm hover:bg-gray-600">View Details</button>` : ''
                    }
                </div>
            `).join('');
        } catch (error) {
            showToast('Error loading tasks', 'error');
        }
    }
    
    async function loadMyTasks() {
        try {
            const response = await fetch('/api/tasks/my-tasks', { headers: { 'Authorization': `Bearer ${jwtToken}` } });
            const tasks = await response.json();
            const container = document.getElementById('myTasksList');
            
            if (tasks.length === 0) {
                container.innerHTML = '<p class="col-span-full text-center text-gray-500">No tasks assigned</p>';
                return;
            }
            
            container.innerHTML = tasks.map(task => `
                <div class="task-card bg-white p-5">
                    <div class="flex justify-between items-start mb-3">
                        <h3 class="font-bold">${escapeHtml(task.title)}</h3>
                        <span class="status-badge status-${task.status}">${task.status}</span>
                    </div>
                    <p class="text-sm text-gray-600 mb-3">${escapeHtml(task.description?.substring(0, 100) || 'No description')}</p>
                    ${task.status === 'IN_PROGRESS' ?
                        `<button onclick="showCompleteModal(${task.id})" class="w-full bg-blue-500 text-white py-2 rounded-lg text-sm hover:bg-blue-600">Complete Task</button>` :
                        task.status === 'COMPLETED' ?
                        `<button onclick="viewTaskDetails(${task.id})" class="w-full bg-gray-500 text-white py-2 rounded-lg text-sm hover:bg-gray-600">View Details</button>` : ''
                    }
                </div>
            `).join('');
        } catch (error) {
            showToast('Error loading my tasks', 'error');
        }
    }
    
    async function takeTask(taskId, btn) {
        const originalText = btn.innerHTML;
        btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Taking...';
        btn.disabled = true;
        
        try {
            const response = await fetch(`/api/tasks/${taskId}/take`, {
                method: 'PUT',
                headers: { 'Authorization': `Bearer ${jwtToken}` }
            });
            
            if (response.ok) {
                showToast('Task taken successfully!', 'success');
                loadTasks();
                loadMyTasks();
                loadDashboard();
            } else {
                showToast('Failed to take task', 'error');
                btn.innerHTML = originalText;
                btn.disabled = false;
            }
        } catch (error) {
            showToast('Error: ' + error.message, 'error');
            btn.innerHTML = originalText;
            btn.disabled = false;
        }
    }
    
    // ========== COMPLETE TASK ==========
    
    let currentTaskId = null;
    
    function showCompleteModal(taskId) {
        currentTaskId = taskId;
        document.getElementById('completeTaskModal').classList.remove('hidden');
        document.getElementById('completeTaskModal').style.display = 'flex';
        document.getElementById('completionDescription').value = '';
    }
    
    function hideCompleteTaskModal() {
        document.getElementById('completeTaskModal').classList.add('hidden');
        document.getElementById('completeTaskModal').style.display = 'none';
        currentTaskId = null;
    }
    
    async function submitCompleteTask() {
        const description = document.getElementById('completionDescription').value;
        if (!description) {
            showToast('Please enter completion description', 'error');
            return;
        }
        
        const btn = document.getElementById('completeTaskBtn');
        const originalText = btn.innerHTML;
        btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Completing...';
        btn.disabled = true;
        
        try {
            const response = await fetch(`/api/tasks/${currentTaskId}/complete?completionDescription=${encodeURIComponent(description)}`, {
                method: 'PUT',
                headers: { 'Authorization': `Bearer ${jwtToken}` }
            });
            
            if (response.ok) {
                showToast('Task completed successfully!', 'success');
                hideCompleteTaskModal();
                loadTasks();
                loadMyTasks();
                loadDashboard();
            } else {
                showToast('Failed to complete task', 'error');
            }
        } catch (error) {
            showToast('Error: ' + error.message, 'error');
        } finally {
            btn.innerHTML = originalText;
            btn.disabled = false;
        }
    }
    
    // ========== CREATE TASK ==========
    
    function showCreateTaskModal() {
        document.getElementById('createTaskModal').classList.remove('hidden');
        document.getElementById('createTaskModal').style.display = 'flex';
    }
    
    function hideCreateTaskModal() {
        document.getElementById('createTaskModal').classList.add('hidden');
        document.getElementById('createTaskModal').style.display = 'none';
        document.getElementById('taskTitle').value = '';
        document.getElementById('taskDescription').value = '';
    }
    
    async function createTask() {
        const title = document.getElementById('taskTitle').value.trim();
        if (!title) {
            showToast('Please enter task title', 'error');
            return;
        }
        
        const taskData = {
            title: title,
            description: document.getElementById('taskDescription').value,
            priority: document.getElementById('taskPriority').value,
            dueDate: null
        };
        
        const btn = event.target;
        const originalText = btn.innerHTML;
        btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Creating...';
        btn.disabled = true;
        
        try {
            const response = await fetch('/api/tasks', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${jwtToken}`
                },
                body: JSON.stringify(taskData)
            });
            
            if (response.ok) {
                showToast('Task created successfully!', 'success');
                hideCreateTaskModal();
                loadTasks();
                loadDashboard();
            } else {
                showToast('Failed to create task', 'error');
            }
        } catch (error) {
            showToast('Error: ' + error.message, 'error');
        } finally {
            btn.innerHTML = originalText;
            btn.disabled = false;
        }
    }
    
    async function viewTaskDetails(taskId) {
        try {
            const response = await fetch(`/api/tasks/${taskId}`, {
                headers: { 'Authorization': `Bearer ${jwtToken}` }
            });
            const task = await response.json();
            
            let details = `Title: ${task.title}\n`;
            details += `Description: ${task.description || 'No description'}\n`;
            details += `Status: ${task.status}\n`;
            details += `Priority: ${task.priority}\n`;
            if (task.completionDescription) {
                details += `\nCompletion Description: ${task.completionDescription}\n`;
            }
            if (task.completedAt) {
                details += `Completed At: ${new Date(task.completedAt).toLocaleString()}\n`;
            }
            
            alert(details);
        } catch (error) {
            showToast('Error loading task details', 'error');
        }
    }
    
    // ========== NAVIGATION ==========
    
    function showSection(section) {
        document.querySelectorAll('.section').forEach(el => el.classList.add('hidden'));
        document.getElementById(`${section}Section`).classList.remove('hidden');
        
        if (section === 'dashboard') loadDashboard();
        if (section === 'tasks') loadTasks();
        if (section === 'myTasks') loadMyTasks();
    }
    
    function escapeHtml(text) {
        if (!text) return '';
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
</script>
